
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Helper function to check if a user is an admin
    function isAdmin() {
      return getUserData(request.auth.uid).role == 'admin';
    }

    // Helper function to get user data from the 'users' collection
    function getUserData(userId) {
      return get(/databases/$(database)/documents/users/$(userId)).data;
    }

    // Rules for the 'users' collection
    match /users/{userId} {
      // Admins can read/write any user profile
      // Students can only read their own profile
      allow read: if isAdmin() || request.auth.uid == userId;
      // Admins can create/update any user
      allow create, update: if isAdmin();
      // Deleting users is restricted to admins
      allow delete: if isAdmin();
    }

    // Rules for the 'exams' collection
    match /exams/{examId} {
      // Admins can perform any operation on exams
      allow read, write, create, delete: if isAdmin();
      
      // Students can only read exams they are assigned to
      allow read: if !isAdmin() && request.auth.uid in resource.data.allowedStudentIds;
    }

    // Rules for the 'submissions' collection
    match /submissions/{submissionId} {
      // Admins can read any submission for grading or review
      allow read: if isAdmin();

      // Students can:
      // 1. Create a submission for an exam they are assigned to.
      // 2. Read their own past submissions.
      allow create: if !isAdmin() && request.auth.uid == request.resource.data.studentId;
      allow read: if !isAdmin() && request.auth.uid == resource.data.studentId;

      // No one should update or delete submissions to maintain record integrity
      allow update, delete: if false;
    }
  }
}
